100以内奇偶数和

#include
using namespace std;
int main()
{
	int j = 0, ou = 0;
	for (int i = 1; i < 100; i++)
	{
		if (i % 2 == 1)
		{
			j+=i;
		}
		else
		{
			ou+=i;
		}
	}
	return 0;
}
